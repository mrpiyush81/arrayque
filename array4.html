function findDuplicateElements(arr) { const seen = []; const duplicates = []; for (const subArray of arr) { const stringifiedSubArray = JSON.stringify(subArray); if (seen.includes(stringifiedSubArray)) { duplicates.push(JSON.parse(stringifiedSubArray)); } else { seen.push(stringifiedSubArray); } } return duplicates; } // Example usage: const a = [[1,2],[1,2],[2,3],[2,3],[1,3]]; const result = findDuplicateElements(a); console.log(result);
